21class PhysicalSystem(Serializable, mm.System): 

22 """ 

23 A physical system for extended phase-space simulations with OpenMM. 

24 

25 Example 

26 ------- 

27 >>> import openxps as xps 

28 >>> from math import pi 

29 >>> import cvpack 

30 >>> from openmm import unit 

31 >>> from openmmtools import testsystems 

32 >>> model = testsystems.AlanineDipeptideVacuum() 

33 >>> umbrella_potential = cvpack.MetaCollectiveVariable( 

34 ... f"0.5*kappa*min(delta,{2*pi}-delta)^2; delta=abs(phi_cv-phi_dv)", 

35 ... [cvpack.Torsion(6, 8, 14, 16, name="phi_cv")], 

36 ... unit.kilojoule_per_mole, 

37 ... kappa=1000 * unit.kilojoule_per_mole / unit.radian**2, 

38 ... phi_dv=0*unit.radian, 

39 ... ) 

40 >>> umbrella_potential.addToSystem(model.system) 

41 >>> phi_dv = xps.ExtraDOF( 

42 ... "phi_dv", 

43 ... unit.radian, 

44 ... 3 * unit.dalton*(unit.nanometer/unit.radian)**2, 

45 ... xps.bounds.Periodic(-180, 180, unit.degree) 

46 ... ) 

47 >>> xps_system = xps.PhysicalSystem(model.system, [phi_dv]) 

48 >>> for xdof in xps_system.extra_dofs: 

49 ... print(xdof) 

50 ExtraDOF(name='phi_dv', unit=rad, mass=3 nm**2 Da/(rad**2), ...) 

51 """ 

52 

53 def __init__( # pylint: disable=super-init-not-called 

54 self, 

55 openmm_system: mm.System, 

56 extra_dofs: t.Iterable[ExtraDOF], 

57 ) -> None: 

58 mmswig.System_swiginit(self, copy(openmm_system).this) 

59 self.extra_dofs = tuple(extra_dofs) 

60 

61 # List all forces that depend on each extra degree of freedom 

62 parameter_forces = defaultdict(list) 

63 for force in self.getForces(): 

64 if hasattr(force, "getNumGlobalParameters"): 

65 for index in range(force.getNumGlobalParameters()): 

66 parameter_forces[force.getGlobalParameterName(index)].append(force) 

67 

68 # Check if all extra degrees of freedom are present in the system's forces 

69 absent_variables = [ 

70 xdof.name for xdof in extra_dofs if xdof.name not in parameter_forces 

71 ] 

72 if absent_variables: 

73 raise ValueError( 

74 "These global parameters are not present in the system's forces: " 

75 + ", ".join(absent_variables) 

76 ) 

77 

78 # Add missing derivatives with respect to the extra degrees of freedom 

79 for xdof in extra_dofs: 

80 for force in parameter_forces[xdof.name]: 

81 if not any( 

82 force.getEnergyParameterDerivativeName(index) == xdof.name 

83 for index in range(force.getNumEnergyParameterDerivatives()) 

84 ): 

85 force.addEnergyParameterDerivative(xdof.name) 

86 

87 def __getstate__(self): 

88 return { 

89 "openmm_system": mm.XmlSerializer.serialize(self), 

90 "extra_dofs": self.extra_dofs, 

91 } 

92 

93 def __setstate__(self, state): 

94 self.__init__( 

95 mm.XmlSerializer.deserialize(state["openmm_system"]), 

96 state["extra_dofs"],
